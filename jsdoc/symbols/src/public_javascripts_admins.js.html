 40 	addHandler : function(id, name) {

 42 			if (event.which == 13) {
 43 				// console.log("Enter-key - save ID:"+id);
 44 				$('#saveform' + id).click();
 45 			}
 46 			if (event.which == 27) { // escape key pressed

 48 				breakEdit(id);
 49 			}
 50 		});
 51 	},

249 	/**
250 	 * replaces a regular (not in edit mode / new user mode) user Informations
251 	 * with the given data
252 	 * 
253 	 * @param {number}
254 	 *            id
255 	 * @param {string}
256 	 *            alias
257 	 * @param {string}
258 	 *            email
259 	 * @param {string}
260 	 *            password
261 	 * @param {???}
262 	 *            adminenabled
263 	 * @param {???}
264 	 *            adminchecked
265 	 * @param {string}
266 	 *            functionstring
267 	 * 
268 	 * @returns {undefined}
269 	 */
270 	replace : function(id, alias, email, password, adminenabled, adminchecked,
271 			functionstring) {
272 		$('#' + id + 'alias').html(alias);
273 		$('#' + id + 'email').html(email);
274 		$('#' + id + 'password').html(password);
275 		$('#' + id + 'function')

277 						functionstring
278 								+ '<img style="visibility:hidden; text-align:right;" id="'
279 								+ id
280 								+ 'wait" height="25px" src="@(routes.Assets.at("images/warte.gif"))');
281 
282 		if (adminenabled)
283 			$('#' + id + 'admininput').removeAttr("disabled");
284 		else
285 			$('#' + id + 'admininput').attr("disabled", "disabled");
286 		if (adminchecked || adminchecked === "true") {
287 			$('#' + id + 'admininput').prop("checked", true);
288 		} else
289 			$('#' + id + 'admininput').removeAttr("checked");
290 
291 		this.updateTable();
