 23 	/**
 24 	 * add enter and escape shortcuts to an arbitrary html-tag to quickly access
 25 	 * save an cancel of the input.
 26 	 * 
 27 	 * @param {number|string}
 28 	 *            id the id of the row that has to be saved or canceled
 29 	 * @param {string}
 30 	 *            name the html-id of the tag that gets shortcut handlers
 31 	 * @returns {undefined}
 32 	 */
 33 	addHandler : function(id, name) {
 34 		$('#' + name).keyup(function(event) { // enter key pressed
 35 			if (event.which == 13) {
 36 				// console.log("Enter-key - save ID:"+id);
 37 				$('#saveform' + id).click();
 38 			}
 39 			if (event.which == 27) { // escape key pressed
 40 				// console.log("Escape-key - break ID:"+id);
 41 				breakEdit(id);
 42 			}
 43 		});
 44 	},
