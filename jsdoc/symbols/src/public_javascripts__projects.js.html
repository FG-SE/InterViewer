 99 	 * Remove the selected user from the select which contains the members of
100 	 * the project
101 	 * 
102 	 * @param {String}
103 	 *            selectForeignId the id of the select with the users that are

111 	modalRemoveSelectedUsers : function(selectForeignId, selectMemberId) {
112 		removeUsers = this.getDataFromSelect($(selectMemberId + " option"),
113 				false);
114 		ExistingUsers = this.getDataFromSelect($(selectForeignId + " option"),
115 				true);
116 
117 		$.each(removeUsers, function() {
118 			if (this.Selected == true) {
119 				var name = this.Name;
120 				$.each(ExistingUsers, function(nr, user) {
121 					if (this.Name === name)
122 						name = null
123 				});
124 				if (name != null)
125 					$(selectForeignId).append(
126 							'<option value="' + this.Value + '">' + this.Name
127 									+ '</option>');
128 				$(selectMemberId + ' option[value=' + this.Value + ']')
129 						.remove();
130 			}
131 		});

171 	/**
172 	 * Collect user data from a html select
173 	 * 
174 	 * @param $select
175 	 *            jquery object of the select
176 	 * @param allUsersSelected
177 	 *            optional parameter true - set selected state of all options
178 	 *            always true false - read selected attribute of the options
179 	 * 
180 	 * @returns {Json}
181 	 */
182 	getDataFromSelect : function($select, allUsersSelected) {
183 		allUsersSelected = allUsersSelected || false;
184 		var data = $select.map(function() {
185 			var object = new Object();
186 			object["Name"] = $(this).text()
187 			object["Value"] = $(this).val();
188 			object["Selected"] = allUsersSelected || $(this).is(":selected");
189 			return object;
190 		}).get();
191 
192 		return data;
193 	},

392 	/**
393 	 * Set a click listener on the property add button to add property to the
394 	 * select
395 	 * 
396 	 * @param {jQuery-object}
397 	 *            $projectEditModal the edit modal
398 	 * @param {jQuery-object}
399 	 *            $select the select
400 	 */
401 	setPropertyAddEvent : function($projectEditModal, $select) {
402 		// Get property name and put it in the property table
403 		$projectEditModal.find("#saveProperty").click(
404 				function() {
405 					var $input = $(this).parent().parent().find("input");
406 					var val = $input.val().trim();
407 
408 					// check whether the text field is empty
409 					if (val.length > 0) {
410 						var duplicate = false;
411 						// true if a hidden element was found
412 						var hiddenProperty = false;
413 
414 						// search for a occurence of the same property in the
415 						// list
416 						$select.find('option').each(
417 								function() {
418 
419 									if ($(this).val() == val)
420 										duplicate = true;
421 
422 									// fade the property in if it was removed
423 									// from the list before
424 									if (!$(this).is(":visible")
425 											&& $(this).val() == val) {
426 										$(this).show();
427 										$input.val('').focus();
428 										duplicate = true;
429 										hiddenProperty = true;
430 										return false;
431 									}
432 								});
433 
434 						if (!duplicate && !hiddenProperty) {
435 							// if the property is not found in the list, add it
436 							// to the list
437 							$select.append(
438 									"<option value='" + val + "'>" + val
439 											+ "</option>").fadeIn(1000);
440 							$input.val('').focus();
441 						} else if (duplicate && !hiddenProperty) {
442 							// display error to sign the already existing
443 							// property
444 							$input.parent().parent().addClass("has-error")
445 									.select();
446 							$input.parent().parent().on('change', function() {
447 								$(this).removeClass("has-error");
448 								$(this).off("change");
449 							});
450 						}
451 					}
452 				});
453 	},

459 	/**
460 	 * Send given project data to server to create a new project and add a new
461 	 * to the table
462 	 * 
463 	 * @param {Json}
464 	 *            project the project data
465 	 * @returns {undefined}
466 	 */
467 	sendAddProject : function(project) {
468 
469 		var _this = this;
470 
471 		// send data to server
472 		$
473 				.ajax({
474 					type : "POST",
475 					url : this.pathPostAddProject,
476 					contentType : "application/json; charset=utf-8",
477 					dataType : "json",
478 					data : JSON.stringify(project),
479 					statusCode : {
480 						200 : function(data) {
481 							// integrate the new project instantly in the list
482 							_this.addRow(data);
483 						},
484 						400 : function(data) {
485 							console.error(data.responseText);
486 							generalAlert("Die Aktion konnte nicht ausgeführt werden: "
487 									+ data.responseText);
488 						}
489 					}
490 				});
491 	},
