1 /**
  2  * @fileOverview: JavaScript file which contains the logic for the
  3  *                interviews-view.
  4  * 
  5  * @author
  6  * @namespace
  7  */
  8 var interviews = {
  9 
 10 	/**
 11 	 * Called on startup: Set event listeners and do initializations
 12 	 * 
 13 	 * @returns {undefined}
 14 	 */
 15 	init : function() {
 16 		/**
 17 		 * Adding an interview, handle user input
 18 		 */
 19 		$("#interviewAddButton").click(
 20 				function() {
 21 					propertytypes['headline'] = "Interview anlegen";
 22 					var interviewAddHTML = Mustache.to_html($(
 23 							'#interviewAddTemplate').html(), propertytypes);
 24 					$("#modalContainer").html(interviewAddHTML);
 25 					$("#interviewAdd").modal("show");
 26 
 27 					$("#interviewAdd").find("#saveInterview").click(
 28 							function() {
 29 								var formData = new FormData(document
 30 										.getElementById("interviewForm"));
 31 								$(this).button("loading");
 32 								interviews.addInterview(formData);
 33 							});
 34 				});
 35 
 36 		/**
 37 		 * Removing an interview, handle user input
 38 		 */
 39 		$("body")
 40 				.on(
 41 						"click",
 42 						"button.interviewRemove",
 43 						function() {
 44 							var _this = this;
 45 
 46 							showConfirmModal(
 47 									function() {
 48 										interviews.removeInterview($(_this)
 49 												.data("interviewid"));
 50 
 51 										// to get all the elements of this row
 52 										// and fade them out
 53 										$(_this).parentsUntil("tbody").fadeOut(
 54 												200);
 55 									},
 56 									unescape("M%F6chten Sie wirklich das Interview inklusive aller Aussagen l%F6schen%3F"),
 57 									function() {
 58 									});
 59 						});
 60 
 61 		/**
 62 		 * Editing an interview, handle user input
 63 		 */
 64 		$("body").on(
 65 				"click",
 66 				"button.interviewEdit",
 67 				function() {
 68 					var interviewId = $(this).data("interviewid");
 69 
 70 					// Read all current values, to put them into the form
 71 					$.getJSON("/getInterviewInfo/" + interviewId, function(
 72 							result) {
 73 
 74 						// create the modal with the form with the current
 75 						// values
 76 						result['headline'] = "Interview bearbeiten";
 77 						var interviewAddHTML = Mustache.to_html($(
 78 								'#interviewAddTemplate').html(), result);
 79 						$("#modalContainer").html(interviewAddHTML);
 80 						$("#interviewAdd").modal("show");
 81 						$("#interviewAdd").find("#saveInterview").click(
 82 								function() {
 83 
 84 									var formData = new FormData(document
 85 											.getElementById("interviewForm"));
 86 									$(this).button("loading");
 87 									interviews.editInterview(formData);
 88 								});
 89 					});
 90 				});
 91 
 92 		/**
 93 		 * Showing interview info
 94 		 */
 95 		$("body").on(
 96 				"click",
 97 				"button.interviewInfo",
 98 				function() {
 99 					$.getJSON("/getInterviewInfo/"
100 							+ $(this).data('interviewid'), function(result) {
101 
102 						var interviewInfoHTML = Mustache.to_html($(
103 								'#interviewInfoTemplate').html(), result);
104 						$("#modalContainer").html(interviewInfoHTML);
105 						$("#interviewInfo").modal("show");
106 					});
107 				});
108 	},
109 
110 	/**
111 	 * Server request to edit the interview with the id
112 	 * @param {object}
113 	 *            formData information to be send to the server
114 	 * @returns {undefined}
115 	 */
116 	editInterview : function(formData) {
117 		$.ajax({
118 			url : $('#ajaxroutes').data("editinterview"),
119 			type : "POST",
120 			data : formData,
121 			processData : false, // tell jQuery not to process the data
122 			contentType : false, // tell jQuery not to set contentType
123 			statusCode : {
124 				200 : function(result) {
125 					var jsonData = $.parseJSON(result);
126 					var interviewRowHTML = Mustache.to_html($(
127 							'#interviewRowTemplate').html(), jsonData);
128 					$("#" + jsonData.interviewId).replaceWith(interviewRowHTML);
129 					$("#interviewAdd").modal("hide");
130 				},
131 				400 : function(result) {
132 					$("#interviewAdd").modal("hide");
133 					generalAlert(result.responseText);
134 				},
135 				500 : function(result) {
136 					$("#interviewAdd").modal("hide");
137 					generalAlert("Interner Serverfehler");
138 				},
139 			}
140 		});
141 	},
142 
143 	/**
144 	 * Server request to add an interview with the data of the form
145 	 * 
146 	 * @param {object}
147 	 *            formData information to be send to the server
148 	 * 
149 	 * @returns {undefined}
150 	 */
151 	addInterview : function(formData) {
152 		$.ajax({
153 			url : $('#ajaxroutes').data("addinterview"),
154 			type : "POST",
155 			data : formData,
156 			processData : false, // tell jQuery not to process the data
157 			contentType : false, // tell jQuery not to set contentType
158 			statusCode : {
159 				200 : function(result) {
160 					$("#interviewAdd").modal("hide");
161 					interviews.addInterviewRow(result);
162 				},
163 				400 : function(result) {
164 					$("#interviewAdd").modal("hide");
165 					generalAlert(result.responseText);
166 				},
167 				500 : function(result) {
168 					$("#interviewAdd").modal("hide");
169 					generalAlert("Interner Serverfehler");
170 				},
171 			}
172 		});
173 	},
174 
175 	/**
176 	 * Server request to remove the interview.
177 	 * 
178 	 * @param {number}
179 	 *            id the id of the interview that will be deleted
180 	 * 
181 	 * @returns {undefined}
182 	 */
183 	removeInterview : function(Id) {
184 		data = {
185 			interviewId : Id
186 		};
187 		$
188 				.ajax({
189 					type : "POST",
190 					url : $('#ajaxroutes').data("removeinterview"),
191 					contentType : "application/json; charset=utf-8",
192 					dataType : "json",
193 					data : JSON.stringify(data),
194 					statusCode : {
195 						200 : function(data) {
196 							console.log("interview gelöscht!");
197 						},
198 						400 : function(data) {
199 							console.error(data.responseText);
200 							generalAlert("Die Aktion konnte nicht ausgeführt werden: "
201 									+ data.responseText);
202 						}
203 					}
204 				});
205 	},
206 
207 	/**
208 	 * Adds an interview to the interview table
209 	 * 
210 	 * @param {jsonstring}
211 	 *            interview the interview data
212 	 * 
213 	 * @returns {undefined}
214 	 */
215 	addInterviewRow : function(interview) {
216 		var json = $.parseJSON(interview);
217 		var interviewRowHTML = Mustache.to_html($('#interviewRowTemplate')
218 				.html(), json);
219 		$("#interviewsTable > tbody").append(interviewRowHTML);
220 	},
221 
222 };
